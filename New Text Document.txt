        function runTool(toolNumber) {
            let inputField = document.getElementById('input' + toolNumber);
            let fileField = document.getElementById('file' + toolNumber);

            if (inputField && inputField.value.trim() === "" && (!fileField || !fileField.value)) {

                return;

            document.getElementById('tool' + toolNumber).style.display = 'none';

            setTimeout(() => {
                document.getElementById('spinner' + toolNumber).style.display = 'none';
                document.getElementById('res' + toolNumber).style.display = 'block';
            }, 2500);
        }

        function resetTool(toolNumber) {
            document.getElementById('res' + toolNumber).style.display = 'none';
            document.getElementById('tool' + toolNumber).style.display = 'flex';

            if (toolNumber === 1) {
                document.getElementById('recordedAudio').style.display = 'none';
                document.getElementById('recordedAudio').src = '';

        function startRecording() {
            navigator.mediaDevices.getUserMedia({ audio: true }).then(stream => {
                mediaRecorder = new MediaRecorder(stream);
                audioChunks = [];
                
                mediaRecorder.ondataavailable = e => {
                    audioChunks.push(e.data);
                };
                
                mediaRecorder.onstop = () => {
                    const blob = new Blob(audioChunks, { 'type': 'audio/ogg; codecs=opus' });
                    const audioUrl = URL.createObjectURL(blob);
                    const audio = document.getElementById('recordedAudio');
                    audio.src = audioUrl;
                    audio.style.display = 'block';
                    document.getElementById('btnDownload').style.display = 'block';
                    
                    stream.getTracks().forEach(track => track.stop());
                };
                
                mediaRecorder.start();
                
                document.getElementById('btnStartRec').disabled = true;
                document.getElementById('btnStopRec').disabled = false;
                document.getElementById('recStatus').textContent = currentLang === 'ar' ? "جاري التسجيل..." : "Recording...";
                document.getElementById('recStatus').style.color = "var(--accent-color)";
                
                totalSeconds = 0;
                clearInterval(timerInterval);
                timerInterval = setInterval(setTimer, 1000);
            }).catch(() => {
                alert(currentLang === 'ar' ? "تعذّر الوصول إلى الميكروفون، يرجى التحقق من الأذونات." : "Unable to access the microphone, please check permissions.");
            });
        }

        function setTimer() {
            ++totalSeconds;
            let hour = Math.floor(totalSeconds / 3600);
            let minute = Math.floor((totalSeconds - hour * 3600) / 60);
            let seconds = totalSeconds - (hour * 3600 + minute * 60);
            
            document.getElementById('recTimer').textContent = (minute < 10 ? "0" + minute : minute) + ":" + (seconds < 10 ? "0" + seconds : seconds);
        }

        function stopRecording() {
            if (mediaRecorder && mediaRecorder.state !== "inactive") {
                mediaRecorder.stop();
                clearInterval(timerInterval);
                document.getElementById('btnStartRec').disabled = false;
                document.getElementById('btnStopRec').disabled = true;
                document.getElementById('recStatus').textContent = currentLang === 'ar' ? "متوقف" : "Stopped";
                document.getElementById('recStatus').style.color = "var(--text-secondary)";
            }
        }

        function downloadRecording() {
            const audio = document.getElementById('recordedAudio');
            if (audio.src) {
                const a = document.createElement('a');
                a.href = audio.src;
                a.download = 'recorded-audio.ogg';
                document.body.appendChild(a);
                a.click();
                document.body.removeChild(a);
            }
        }

        function togglePlay(btn) {
            let icon = btn.querySelector('i');
            if (icon.classList.contains('fa-play')) {
                icon.classList.remove('fa-play');
                icon.classList.add('fa-pause');
            } else {
                icon.classList.remove('fa-pause');
                icon.classList.add('fa-play');
            }
        }

        function filterEpisodes(category) {
            const buttons = document.querySelectorAll('.filter-btn');
            buttons.forEach(btn => btn.classList.remove('active'));
            
            const ev = window.event;
            if (ev && ev.target) {
                ev.target.classList.add('active');
            }

            const cards = document.querySelectorAll('.episode-card');
            cards.forEach(card => {
                if (category === 'all' || card.getAttribute('data-category') === category) {
                    card.style.display = 'block';
                } else {
                    card.style.display = 'none';
                }
            });
        }
